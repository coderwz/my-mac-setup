## git config

```shell
[alias]
      co = checkout
      br = branch
      ci = commit
      st = status
      hs = log --pretty='%C(yellow)%h %C(cyan)%cd %Cblue%aN%C(auto)%d %Creset%s' --graph --date=relative --date-order
      amend = commit --amend --no-edit
      am = commit --amend --no-edit
      cp = cherry-pick
```
